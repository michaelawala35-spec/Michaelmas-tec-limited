    <script>
        let display = document.getElementById('display');
        let currentInput = '0';

        function updateDisplay() { display.textContent = currentInput; }
        function clearDisplay() { currentInput = '0'; updateDisplay(); }
        
        function appendToDisplay(v) { 
            if (currentInput === '0' && v !== '.') currentInput = v; 
            else currentInput += v; 
            updateDisplay(); 
        }

        function closeVault() { document.getElementById('vault').style.display = 'none'; }

        function calculate() {
            // SECRET ACCESS CODE
            if (currentInput === '9999') {
                document.getElementById('vault').style.display = 'flex';
                currentInput = '0';
                updateDisplay();
                return;
            }
            try {
                let result = eval(currentInput.replace('×', '*').replace('÷', '/'));
                currentInput = Number.isInteger(result) ? result.toString() : result.toFixed(2).toString();
                updateDisplay();
            } catch {
                currentInput = 'Error';
                updateDisplay();
            }
        }

        document.getElementById('fileInput').onchange = function(e) {
            let reader = new FileReader();
            reader.onload = function() {
                let img = document.getElementById('secretImage');
                img.src = reader.result;
                img.style.display = 'block';
            }
            reader.readAsDataURL(e.target.files[0]);
        }
    </script>
